
### 1.3 核心挑战

1. **Game.ts 过于庞大**：2400行，所有场景渲染集中在一处
2. **HTML/DOM渲染**：现有UI基于innerHTML模板字符串
3. **状态管理**：PlayerData使用简单对象，需扩展新数值
4. **低耦合要求**：新系统不能破坏现有架构

```typescript
class EventManager {
  checkDailyEvent(day: number, state: PlayerState): GameEvent | null {
    // 1. 检查剧情节点
    const storyNode = this.getStoryNode(day);
    if (storyNode) return storyNode;
    
    // 2. 30%概率随机事件
    if (Math.random() < 0.3) {
      return this.getRandomEvent(state);
    }
    
    return null;
  }
}
```

```typescript
async function dailyFlow() {
  // 1. 白天：现实事件
  const survivalResult = survivalSystem.processDaily();
  if (survivalResult.crisis) {
    // 处理危机事件
  }
  
  // 2. 检查NPC互动
  const npcEvent = npcSystem.checkDailyInteraction(day);

    // 显示NPC事件

  // 4. 触发事件
  const event = eventManager.checkDailyEvent(day, playerState);
  if (event) {
    // 显示事件选项

    applyChoiceEffects(choice);
    
    // 5. 生成热搜（重大事件）

    }
  }
  
  // 6. 下播结算
  renderDailySummaryWithTrending();
}
```

```typescript
// 使用自定义事件
class GameEventBus extends EventTarget {
  emit(event: string, data: any): void {
    this.dispatchEvent(new CustomEvent(event, { detail: data }));
  }
  
  on(event: string, handler: (e: CustomEvent) => void): void {
    this.addEventListener(event, handler as EventListener);
  }
}

export const eventBus = new GameEventBus();

// 使用示例
// NPCSystem.ts
eventBus.emit('npc:relationChanged', { npcId: 'landlady', delta: 30 });

// Game.ts

  // 更新UI显示
});
```

## 五、游戏流程实现

### 5.1 每日流程代码

```typescript
// 在Game.ts中修改每日流程
async function dailyFlow() {
  // 1. 开场独白
  await this.renderDailyOpening();
  
  // 2. 白天：生存检查
  const survivalResult = survivalSystem.processDaily();
  if (survivalResult.crisis) {
    await this.renderCrisisEvent(survivalResult.crisis);
  }
  
  // 3. NPC互动检查
  const npcEvent = npcSystem.checkDailyInteraction();
  if (npcEvent) {
    await this.renderNPCDialog(npcEvent.npcId);
  }
  
  // 4. 晚上：直播（保留原有分区选择）
  await this.renderCategorySelect();  // 原有方法
  
  // 5. 直播中进行弹幕互动
  danmakuSystem.start();
  
  // 6. 检查事件
  const event = eventManager.checkDailyEvent(day, playerState);
  if (event) {
    await this.renderDoubleLifeEvent(event);
    const choice = await this.waitForChoice();
    this.applyChoiceEffects(choice);
    
    // 7. 生成热搜
    if (event.isMajor) {
      const hotSearch = hotSearchSystem.generate(event);
      await this.renderHotSearch(hotSearch);
    }
  }
  
  // 8. 下播结算
  await this.renderDailySummaryWithSurvival();
}
```
